void myMsDelay (unsigned int time)
{
	unsigned int i, j;
	for(i=0; i<time; i++)
	{
	    for(j=0; j<710; j++);
	}
}
void PORTs_init(void)
{
	INTCON2bits.RBPU = 0;
	TRISB = 0X0F;
	TRISD = 0x00;
	TRISE = 0x00;
	ADCON1 = 0x0E;
}

void display_string_LCD(static char *pstring1)
{
	int count=0, len;
	TRISE = 0x00;
	TRISD = 0x00;
	LCD_write_string(pstring1);
	myMsDelay(15);
}
void init_LCD(void)
{
	LCD_cmd(0x38);
	myMsDelay(15);
	LCD_cmd(0x01);
	myMsDelay(15);
	LCD_cmd(0x0C);
	myMsDelay(15);
	LCD_cmd(0x80);
	myMsDelay(15);
}
void LCD_cmd(unsigned char ord)
{
	LCD_DATA = ord;
	rs = 0;
	rw = 0;
	en = 1;
	myMsDelay(15);
	en = 0;
	myMsDelay(15);
}
void LCD_write(unsigned char data)
{
	LCD_DATA = data;
	rs = 1;
	rw = 0;
	en = 1;
	myMsDelay(15);
	en = 0;
	myMsDelay(15);
}
void LCD_write_string(static char *str)
{
	int i = 0;
	while(str[i] != 0)
	{
		LCD_write(str[i]);
		myMsDelay(15);
		i++;
	}
}
